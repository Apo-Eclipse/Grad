DATABASE SCHEMA (based on Django models):

TABLE: core_transaction
  - id (bigint, PK)
  - user_id (bigint, FK -> auth_user.id)
  - budget_id (bigint, FK -> core_budget.id)
  - date (date)
  - amount (numeric(12,2))
  - time (time, nullable)
  - store_name (text, nullable)
  - city (text, nullable)
  - type_spending (text, nullable)
  - neighbourhood (text, nullable)
  - active (boolean, default true)
  - created_at (timestamp with time zone, auto_now_add)
  - updated_at (timestamp with time zone, nullable)

TABLE: core_budget
  - id (bigint, PK)
  - user_id (bigint, FK -> auth_user.id)
  - budget_name (text)
  - description (text, nullable)
  - total_limit (numeric(12,2))
  - priority_level_int (smallint, nullable)
  - active (boolean, default true)
  - created_at (timestamp with time zone, auto_now_add)
  - updated_at (timestamp with time zone, nullable)

TABLE: core_income
  - id (bigint, PK)
  - user_id (bigint, FK -> auth_user.id)
  - type_income (text)
  - amount (numeric(12,2), nullable)
  - description (text, nullable)
  - active (boolean, default true)
  - created_at (timestamp with time zone, auto_now_add)
  - updated_at (timestamp with time zone, nullable)

TABLE: core_goal
  - id (bigint, PK)
  - user_id (bigint, FK -> auth_user.id)
  - goal_name (text)
  - description (text, nullable)
  - target (numeric(12,2), nullable)
  - start_date (date, nullable)
  - due_date (date, nullable)
  - plan (text, nullable)
  - active (boolean, default true)
  - created_at (timestamp with time zone, auto_now_add)
  - updated_at (timestamp with time zone, nullable)

TABLE: core_profile (One-to-One with auth_user)
  - id (bigint, PK)
  - user_id (bigint, FK, Unique -> auth_user.id)
  - job_title (text, nullable)
  - address (text, nullable)
  - birthday (date, nullable)
  - gender (text, nullable, choices)
  - employment_status (text, nullable, choices)
  - education_level (text, nullable, choices)
  - user_persona (jsonb, nullable)
  - created_at (timestamp with time zone, auto_now_add)
  - updated_at (timestamp with time zone, nullable)

TABLE: core_chatconversation
  - id (bigint, PK)
  - user_id (bigint, FK -> auth_user.id)
  - title (text, nullable)
  - channel (text, nullable)
  - summary_text (text, nullable)
  - started_at (timestamp with time zone, auto_now_add)
  - last_message_at (timestamp with time zone, nullable)
  - summary_created_at (timestamp with time zone, nullable)

TABLE: core_chatmessage
  - id (bigint, PK)
  - conversation_id (bigint, FK -> core_chatconversation.id)
  - sender_type (text)
  - source_agent (text)
  - content (text)
  - content_type (text, nullable)
  - language (text, nullable)
  - created_at (timestamp with time zone, auto_now_add)

TABLE: auth_user (Standard Django User)
  - id (bigint, PK)
  - username (text)
  - email (text)
  - first_name (text)
  - last_name (text)
  - ... (other standard auth fields)
